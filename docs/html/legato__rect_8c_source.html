167

173  else

175  miny = p2->y;
176  maxy = p1->y;
177  }
178

205  // rectangles actually have to intersect

211  {
212  res[count].x = rem.x;
213  res[count].y = rem.y;
214  res[count].width = obj->x - rem.x;
215  res[count].height = rem.height;

217  rem.x += res[count].width;
218  rem.width -= res[count].width;

220  count++;
221  }
