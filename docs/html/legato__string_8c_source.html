59 {
60  uint32_t idx;
61  uint32_t len, lines;
62  leRect lineRect;

64 
65  LE_ASSERT_THIS();
66 
67  *rect = leRect_Zero;
68 
69  len = _this->fn->length(_this);
70 
71  if(len == 0)
72  {
73  return LE_SUCCESS;
74  }
75 
76  lines = _this->fn->getLineCount(_this);
77 
78  for(idx = 0; idx < lines; idx++)
79  {

81

91 
92  if(rect->width < lineRect.width)

94  rect->width = lineRect.width;
95  }
96  }
97 
98  return LE_SUCCESS;

102 {

105  uint32_t count;
106  leChar chr;
107 
108  LE_ASSERT_THIS();
109 
110  len = _this->fn->length(_this);
111 
112  count = 1;
113 
114  if(len == 0)
115  return count;
116 
117  for(idx = 0; idx < len; idx++)
118  {

175 leResult _leString_GetLineRect(const leString* _this,
176  uint32_t line,
177  leRect* rect)
178 {
179  uint32_t idx;
180  uint32_t len;
181  leRasterFont* fnt;
182  leFontGlyph glyph;
183  uint32_t startIdx, endIdx;
184  leChar chr;
185 
186  LE_ASSERT_THIS();
187 
188  *rect = leRect_Zero;
189 
190  len = _this->fn->length(_this);
191  fnt = (leRasterFont*)_this->fn->getFont(_this);
192 
193  if(len == 0 || fnt == NULL || line >= _this->fn->getLineCount(_this))
194  return LE_FAILURE;
195 
196  if(fnt->glyphTable == NULL)
197  return LE_FAILURE;
198 
199  _this->fn->getLineIndices(_this, line, &startIdx, &endIdx);
200 
201  for(idx = startIdx; idx < endIdx; idx++)
202  {
203  chr = _this->fn->charAt(_this, idx);
204 
205  leFont_GetGlyphInfo((leFont*)fnt, chr, &glyph);
206 
207  rect->width += glyph.advance;
208  }
209 
210  rect->height = fnt->height;
211 
212  return LE_SUCCESS;
213 }

277  LE_ASSERT_THIS();
278 
279  rect.x = 0;
280  rect.y = 0;
281 
282  *charIdx = 0;
283 
284  if(pt->x < 0 || pt->y < 0)
285  return LE_FAILURE;
286 
287  len = _this->fn->length(_this);
288  fnt = (leRasterFont*)_this->fn->getFont(_this);
289 
290  if(len == 0 || fnt == NULL)

292 
293  for(idx = 0; idx < len; idx++)
294  {
295  chr = _this->fn->charAt(_this, idx);
296 
297  if(chr == LE_STRING_LINEBREAK)
298  {
299  rect.x = 0;
300  rect.y += fnt->height;
301  }
302  else

305 
306  rect.width = glyph.advance;

308 
309  if(leRectContainsPoint(&rect, pt) == LE_TRUE)
310  {
311  *charIdx = idx;
312 
313  return LE_SUCCESS;
314  }
315 
316  rect.x += glyph.advance;
317  }
