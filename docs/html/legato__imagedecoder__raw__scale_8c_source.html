42 static leResult stage_scaledNearestNeighborPreRead(struct ScaleNearestNeighborPreReadStage* stage)
43 {
44  stage->base.state->readCount = 1;
45 
46  // calculate nearest neighbor index
47  stage->base.state->readOperation[0].x = (lePercentOf(stage->base.state->referenceX, stage->stretchRatioX) + 50) / 100;
48 
49  if(stage->base.state->readOperation[0].x >= stage->base.state->source->buffer.size.width)
50  {
51  stage->base.state->readOperation[0].x = stage->base.state->source->buffer.size.width - 1;
52  }
53 
54  stage->base.state->readOperation[0].y = (lePercentOf(stage->base.state->referenceY, stage->stretchRatioY) + 50) / 100;
55 
56  if(stage->base.state->readOperation[0].y >= stage->base.state->source->buffer.size.height)
57  {
58  stage->base.state->readOperation[0].y = stage->base.state->source->buffer.size.height - 1;
59  }
60 
61  // calculate source index
62  stage->base.state->readOperation[0].bufferIndex = (stage->base.state->readOperation[0].x) +
63  ((stage->base.state->readOperation[0].y) *
64  stage->base.state->source->buffer.size.width);
65 
66  return LE_SUCCESS;
67 }
