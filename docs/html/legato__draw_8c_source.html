177 
178  // blend with alpha argument value
179  rgbaSource &= ~(RGBA_8888_ALPHA_MASK);
180  rgbaSource |= a;
181 
182  rgbaDest = leColorConvert(LE_GLOBAL_COLOR_MODE, LE_COLOR_MODE_RGBA_8888, nativeSource);
183  rgbaDest |= RGBA_8888_ALPHA_MASK;
184 
185  resultClr = leColorBlend_RGBA_8888(rgbaSource, rgbaDest);
186 
187  // convert to destination format
188  clr = leColorConvert(LE_COLOR_MODE_RGBA_8888, LE_GLOBAL_COLOR_MODE, resultClr);
189

191  x,
