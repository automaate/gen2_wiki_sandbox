47 leResult leImage_Create(leImage* img,
48  uint32_t width,
49  uint32_t height,
50  leColorMode mode,
51  void* data,
52  uint32_t locationID)
53 {
54  if(img == NULL || width == 0 || height == 0)
55  return LE_FAILURE;
56 
57  img->header.address = data;
58  img->header.location = locationID;
59  img->header.size = width * height * leColorInfoTable[mode].size;
60  img->format = LE_IMAGE_FORMAT_RAW;
61  img->buffer.size.width = width;
62  img->buffer.size.height = height;
63  img->buffer.buffer_length = img->header.size;
64  img->buffer.mode = mode;
65  img->buffer.pixels = data;
66  img->buffer.pixel_count = width * height;
67  img->flags = 0;
68  img->mask.color = 0;
69  img->alphaMap = NULL;
70  img->palette = NULL;
71 
72  return LE_SUCCESS;
73 }

75 leImage* leImage_Allocate(uint32_t width,
76  uint32_t height,
77  leColorMode mode)
78 {
79  leImage* img;
80 
81  if(width == 0 || height == 0)
82  return NULL;
83 
84  img = LE_MALLOC(sizeof(leImage));
85 
86  if(img == NULL)
87  return NULL;
88 
89  img->header.size = width * height * leColorInfoTable[mode].size;
90 
91  img->header.address = LE_MALLOC(img->header.size);
92 
93  if(img->header.address == NULL)
94  {
95  leFree(img);
96 
97  return NULL;
98  }
99 
100  img->header.location = LE_STREAM_LOCATION_ID_INTERNAL;
101  img->format = LE_IMAGE_FORMAT_RAW;
102  img->buffer.size.width = width;
103  img->buffer.size.height = height;
104  img->buffer.buffer_length = img->header.size;
105  img->buffer.mode = mode;
106  img->buffer.pixels = img->header.address;
107  img->buffer.pixel_count = width * height;
108  img->flags = LE_IMAGE_INTERNAL_ALLOC;
109  img->mask.color = 0;
110  img->alphaMap = NULL;
111  img->palette = NULL;
112 
113  return img;
114 }

132 {

134  uint32_t decIdx;

137  return LE_FAILURE;
138 
139  for(decIdx = 0; decIdx < MAX_IMAGE_DECODER_COUNT; decIdx++)

142  {
143  if(decoders[decIdx]->draw(img,
144  sourceRect,
145  x,
146  y,
147  a) == LE_SUCCESS)
148  {
149  decoders[decIdx]->exec();
150 
151  break;
152  }
153  }
154  }
155 
156  return LE_SUCCESS;
