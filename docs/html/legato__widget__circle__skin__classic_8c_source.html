56 static void nextState(leCircleWidget* cir)
57 {
58  switch(cir->widget.drawState)
59  {
60  case NOT_STARTED:
61  {
62  paintState.alpha = 255;
63 
64 #if LE_ALPHA_BLENDING_ENABLED == 1
65  if(cir->fn->getCumulativeAlphaEnabled(cir) == LE_TRUE)
66  {
67  paintState.alpha = cir->fn->getCumulativeAlphaAmount(cir);
68  }
69 #endif
70 
71  if(cir->widget.backgroundType != LE_WIDGET_BACKGROUND_NONE)
72  {
73  cir->widget.drawState = DRAW_BACKGROUND;
74  cir->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
75 
76  return;
77  }
78  }
79  case DRAW_BACKGROUND:
80  {
81  cir->widget.drawState = DRAW_CIRCLE;
82  cir->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawCircle;
83 
84  return;
85  }
86  case DRAW_CIRCLE:
87  {
88  if(cir->widget.borderType != LE_WIDGET_BORDER_NONE)
89  {
90  cir->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
91  cir->widget.drawState = DRAW_BORDER;
92 
93  return;
94  }
95  }
96  case DRAW_BORDER:
97  {
98  cir->widget.drawState = DONE;
99  cir->widget.drawFunc = NULL;
100  }
101  }
102 }
