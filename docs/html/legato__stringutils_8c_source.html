607 static leResult decodeUTF8(uint8_t* val,
608  uint32_t max,
609  uint32_t* codePoint,
610  uint32_t* size)
611 {
612  uint32_t point = 0;
613  uint32_t length = 0;
614 
615  if((val[0] & 0x80) == 0x0)
616  {
617  length = 1;
618  point = val[0] & 0x7F;
619  }
620  else if((val[0] & 0xE0) == 0xC0)
621  {
622  length = 2;
623 
624  if(max < 2)
625  return LE_FAILURE;
626 
627  point |= (val[0] & 0x1F) << 6;
628  point |= (val[1] & 0x3F);
629  }
630  else if((val[0] & 0xF0) == 0xE0)
631  {
632  length = 3;
633 
634  if(max < 3)
635  return LE_FAILURE;
636 
637  point |= (val[0] & 0xF) << 12;
638  point |= (val[1] & 0x3F) << 6;
639  point |= (val[2] & 0x3F);
640  }
641  else if((val[0] & 0xF8) == 0xF0)
642  {
643  length = 4;
644 
645  if(max < 4)
646  return LE_FAILURE;
647 
648  point |= (val[0] & 0x7) << 18;
649  point |= (val[1] & 0x3F) << 12;
650  point |= (val[2] & 0x3F) << 6;
651  point |= (val[3] & 0x3F);
652  }
653  else if((val[0] & 0xFC) == 0xF8)
654  {
655  length = 5;
656 
657  if(max < 5)
658  return LE_FAILURE;
659 
660  point |= (val[0] & 0x3) << 24;
661  point |= (val[1] & 0x3F) << 18;
662  point |= (val[2] & 0x3F) << 12;
663  point |= (val[3] & 0x3F) << 6;
664  point |= (val[4] & 0x3F);
665  }
666  else if((val[0] & 0xFE) == 0xFC)
667  {
668  length = 6;
669 
670  if(max < 6)
671  return LE_FAILURE;
672 
673  point |= (val[0] & 0x1) << 30;
674  point |= (val[1] & 0x3F) << 24;
675  point |= (val[2] & 0x3F) << 18;
676  point |= (val[3] & 0x3F) << 12;
677  point |= (val[4] & 0x3F) << 6;
678  point |= (val[5] & 0x3F);
679  }
680  else
681  return LE_FAILURE;
682 
683  *size = length;
684  *codePoint = point;
685 
686  return LE_SUCCESS;
687 }

689 static leResult decodeUTF16(uint8_t* val,
690  uint32_t max,
691  uint32_t* codePoint,
692  uint32_t* size)
693 {
694  uint32_t point = 0;
695  uint32_t high, low;
696 
697  if((val[1] & 0xFC) == 0xDC)
698  {
699  if(max < 4)
700  return LE_FAILURE;
701 
702  *size = 4;
703 
704  high = val[0] << 8;
705  high |= val[1];
706  high -= 0xD800;
707 
708  low = val[2] << 8;
709  low |= val[3];
710  low -= 0xDC00;
711 
712  point = ((high << 10) + low) + 0x10000;
713  }
714  else
715  {
716  if(max < 2)
717  return LE_FAILURE;
718 
719  *size = 2;
720 
721  point += val[0];
722  point += (val[1] << 8);
723  }
724 
725  *codePoint = point;
726 
727  return LE_SUCCESS;
728 }
