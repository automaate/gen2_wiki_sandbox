157 laResult laEvent_ProcessEvents_RTOS()
158 {
159  laContext* context = laContext_GetActive();
160  laEvent* evt;
161  laListNode* node;
162  laEventResult res;
163 
164  if(context == NULL)
165  return LA_FAILURE;
166 
167  if(context->event.events.size == 0)
168  return LA_FAILURE;
169 
170  node = context->event.events.head;
171 
172  if(context->event.events.size > 1)
173  {
174  node = node;
175  }
176 
177  while(node != NULL)
178  {
179  //evt = context->event.events.values[i];
180  evt = (laEvent*)node->val;
181 
182  if(context->event.filter != NULL &&
183  context->event.filter(evt) == LA_FALSE)
184  continue;
185 
186  node = node->next;
187 
188  res = processEvent_RTOS(evt);
189 
190  if(res == LA_EVENT_HANDLED)
191  {
192  laList_Remove(&context->event.events, evt);
193 
194  context->memIntf.heap.free(evt);
195  }
196  else if(res == LA_EVENT_RESET_QUEUE)
197  {
198  laList_Destroy(&context->event.events);
199 
200  node = NULL;
201  }
202  }
203 
204  return LA_SUCCESS;
205 }
