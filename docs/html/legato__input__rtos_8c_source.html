40 laResult laInput_SendTouchDown_Ext_RTOS(uint32_t id, int32_t x, int32_t y)
41 {
42  laInput_TouchDownEvent* evt;
43  laContext* context = laContext_GetActive();
44  GFX_Point pnt;
45 
46  if(context == NULL || // need a valid context
47  context->input.enabled == LA_FALSE || // inputs are disabled
48  id >= LA_MAX_TOUCH_STATES || // don't overrun array
49  context->input.touch[id].valid == LA_TRUE) // touch id must be invalid
50  return GFX_FAILURE;
51 
52  // reorient touch coordinates if the user interface is rotated
53  pnt = transformCoordinate(x, y);
54 
55  // dispatch the event
56  context->input.touch[id].valid = LA_TRUE;
57  context->input.touch[id].x = pnt.x;
58  context->input.touch[id].y = pnt.y;
59 
60  evt = context->memIntf.heap.calloc(1, sizeof(laInput_TouchDownEvent));
61 
62  if(evt == NULL)
63  return GFX_FAILURE;
64 
65  evt->event.id = LA_EVENT_TOUCH_DOWN;
66  evt->touchID = id;
67  evt->x = pnt.x;
68  evt->y = pnt.y;
69 
70 #ifdef INPUT_EVENT_DEBUG
71  printf("adding touch down event %i, %i\n", evt->x, evt->y);
72 #endif
73 
74  if(laEvent_SendEvent_Ext_RTOS((laEvent*)evt) == GFX_FAILURE)
75  {
76  context->memIntf.heap.free(evt);
77 
78  return GFX_FAILURE;
79  }
80 
81  return GFX_SUCCESS;
82 }

125 laResult laInput_SendTouchMoved_Ext_RTOS(uint32_t id, int32_t x, int32_t y)
126 {
127  laInput_TouchMovedEvent* evt;
128  laEvent* evtPtr;
129  laContext* context = laContext_GetActive();
130  GFX_Point pnt;
131  laListNode* node;
132 
133  if(context == NULL || // need a valid context

135  context->input.touch[id].valid == LA_FALSE) // touch id must be valid

138  // find any existing touch moved event and overwrite

140 
141  while(node != NULL)
142  {

144 
145  if(evtPtr->id == LA_EVENT_TOUCH_MOVED)

148 
149  if(evt->touchID == id)
150  {
151 #ifdef INPUT_EVENT_DEBUG
152  printf("overwriting previous move event\n");
153 #endif
154 
155  // reorient touch coordinates if the user interface is rotated
156  pnt = transformCoordinate(x, y);
157 
158  evt->x = x;
159  evt->y = y;
160 
161  evt->x = pnt.x;
162  evt->y = pnt.y;
163 
164  context->input.touch[id].x = pnt.x;
165  context->input.touch[id].y = pnt.y;
166 
167  return LA_SUCCESS;
168  }
169  }
170 
171  node = node->next;
172  }
173 
174  evt = context->memIntf.heap.calloc(1, sizeof(laInput_TouchMovedEvent));
175 
176  if(evt == NULL)
177  return GFX_FAILURE;
178 
179  // reorient touch coordinates if the user interface is rotated
180  pnt = transformCoordinate(x, y);
181 
182  evt->event.id = LA_EVENT_TOUCH_MOVED;
183  evt->touchID = id;
184  evt->prevX = context->input.touch[id].x;

190  context->input.touch[id].y = pnt.y;
191 
192 #ifdef INPUT_EVENT_DEBUG
