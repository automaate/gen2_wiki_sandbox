78 static void nextState(leTouchTestWidget* tch)
79 {
80  switch(tch->widget.drawState)
81  {
82  case NOT_STARTED:
83  {
84  paintState.alpha = 255;
85 
86 #if LE_ALPHA_BLENDING_ENABLED == 1
87  if(tch->fn->getCumulativeAlphaEnabled(tch) == LE_TRUE)
88  {
89  paintState.alpha = tch->fn->getCumulativeAlphaAmount(tch);
90  }
91 #endif
92 
93  if(tch->widget.backgroundType != LE_WIDGET_BACKGROUND_NONE)
94  {
95  tch->widget.drawState = DRAW_BACKGROUND;
96  tch->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
97 
98  return;
99  }
100  }
101  case DRAW_BACKGROUND:
102  {
103  if(tch->size > 0)
104  {
105  tch->widget.drawState = DRAW_LINES;
106  tch->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawLines;
107 
108  return;
109  }
110  }
111  case DRAW_LINES:
112  {
113  if(tch->widget.borderType != LE_WIDGET_BORDER_NONE)
114  {
115  tch->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
116  tch->widget.drawState = DRAW_BORDER;
117 
118  return;
119  }
120  }
121  case DRAW_BORDER:
122  {
123  tch->widget.drawState = DONE;
124  tch->widget.drawFunc = NULL;
125  }
126  }
127 }

136 static void drawLines(leTouchTestWidget* tch)
137 {
138  leRect widgetRect;
139  lePoint touchPnt;
140  uint32_t per;
141  uint32_t i, j;
142  leColor c1, c2, clr;
143 
144  widgetRect = tch->fn->rectToScreen(tch);
145 
146  // draw lines
147  j = tch->start;
148 
149  for(i = 0; i < tch->size; i++)
150  {
151  if(j >= tch->size)
152  {
153  j = 0;
154  }
155 
156  per = lePercentWholeRounded(tch->pnts[j].y, widgetRect.height);
157 
158  c1 = tch->widget.scheme->foreground;
159  c2 = tch->widget.scheme->foregroundInactive;
160 
161  clr = leColorLerp(c1,
162  c2,
163  per,
164  LE_GLOBAL_COLOR_MODE);
165 
166  touchPnt = tch->pnts[j];
167 
168  leRenderer_HorzLine(widgetRect.x,
169  widgetRect.y + touchPnt.y,
170  widgetRect.width,
171  clr,
172  paintState.alpha);
173 
174  per = lePercentWholeRounded(tch->pnts[j].x, widgetRect.width);
175 
176  c1 = tch->widget.scheme->background;
177  c2 = tch->widget.scheme->backgroundInactive;
178 
179  clr = leColorLerp(c1,
180  c2,
181  per,
182  LE_GLOBAL_COLOR_MODE);
183 
184  leRenderer_VertLine(widgetRect.x + touchPnt.x,
185  widgetRect.y,
186  widgetRect.height,
187  clr,
188  paintState.alpha);
189 
190  j++;
191  }
192 
193  nextState(tch);
194 }
