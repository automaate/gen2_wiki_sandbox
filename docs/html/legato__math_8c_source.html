65 {

72 
73  return (-_cosTable[v]);

132 leResult leEllipsePoint(int32_t t, int32_t a, int32_t b, int32_t theta, lePoint* p)
133 {
134  t = leNormalizeAngle(t);
135 
136  p->x = (a * leCos(t) * leCos(theta) / (TRIG_SCALAR * TRIG_SCALAR)) -
137  (b * leSin(t) * leSin(theta) / (TRIG_SCALAR * TRIG_SCALAR));
138 
139  p->y = (a * leCos(t) * leSin(theta) / (TRIG_SCALAR * TRIG_SCALAR)) +
140  (b * leSin(t) * leCos(theta) / (TRIG_SCALAR * TRIG_SCALAR));
141 
142  return LE_SUCCESS;
143 }

391  }
392  default:
393  {
394  return LE_FALSE;
395  }
396  }
397 
398  //Re-orient angles to a CCW orientation
399  if (dir == LE_CW)
400  {
401  angle = startAngleArc0;
402  startAngleArc0 = endAngleArc0;
403  endAngleArc0 = angle;
404  }
405 
406  //if arc wraps around 0/360, do triple arc test
407  if (startAngleArc0 > endAngleArc0)
408  {
409  //split between (start -> 360) and (0 -> end)
410  startTestAngle = startAngleArc0;
411  endTestAngle = 360;
412

416  }

420  }
421

427  {

433  startTestAngle = 0;

435 
436  if (startTestAngle <= quadStartAngle)
437  {
438  baseAngle = 0;
439  }

442  baseAngle = 1;
443  }
444

448  }
449  else if (baseAngle == 1 && (quadEndAngle >= startTestAngle))
450  {
451  retval = LE_TRUE;
452  }
453  }
454  }
455  else if (startAngleArc0 < endAngleArc0)
