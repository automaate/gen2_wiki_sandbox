139 leChar _leTableString_CharAt(const leTableString* _this,
140  uint32_t idx)
141 {
142  leFont* fnt = NULL;
143  uint32_t i, j;
144  uint32_t codePoint;
145  uint32_t offset = 0;
146  const leStringTable* tbl = leGetState()->stringTable;
147  uint32_t lang = leGetState()->languageID;
148  leStringInfo info;
149 
150  LE_ASSERT_THIS();
151 
152  if(tbl == NULL)
153  return 0;
154 
155  fnt = leStringTable_GetStringFont(tbl, _this->index, lang);
156 
157  if(fnt == NULL)
158  return 0;
159 
160  info.stringIndex = _this->index;
161  info.languageID = lang;
162 
163  if(leStringTable_StringLookup(tbl, &info) == LE_FAILURE)
164  return LE_FAILURE;
165 
166  j = 0;
167 
168  for(i = 0; i < info.dataSize;)
169  {
170  if(leDecodeCodePoint(tbl->encodingMode,
171  info.data + i,
172  info.dataSize,
173  &codePoint,
174  &offset) == LE_FAILURE)
175  return LE_FAILURE;
176 
177  i += offset;
178 
179  if(j == idx)
180  return codePoint;
181 
182  j++;
183  }
184 
185  return 0;
186 }

190 int32_t _leTableString_Compare(const leTableString* _this,
191  const leString* tgt)
192 {
193  uint32_t myLen, len, itr, chr;
194 
195  LE_ASSERT_THIS();
196 
197  if(tgt == NULL)
198  return -1;
199 
200  myLen = _this->fn->length(_this);
201  len = tgt->fn->length(tgt);
202 
203  if(myLen < len)
204  {
205  len = myLen;
206  }
207 
208  if(len == 0)
209  return 0;
210 
211  for(itr = 0; itr < len; itr++)
212  {
213  chr = tgt->fn->charAt(tgt, itr);
214 
215  if(_this->fn->charAt(_this, itr) != chr)
216  {
217  return _this->fn->charAt(_this, itr) - chr;
218  }
219  }
220 
221  return 0;
222 }

248 uint32_t _leTableString_ToChar(const leTableString* _this, leChar* buf, uint32_t size)
249 {
250  uint32_t myLen, itr;
251 
252  LE_ASSERT_THIS();
253 
254  if(_this->index == LE_STRING_NULLIDX ||
255  buf == NULL ||
256  size == 0)
257  {
258  return LE_FAILURE;
259  }
260 
261  myLen = _this->fn->length(_this);
262 
263  if(size >= myLen)
264  {
265  size = myLen;
266  }
267 
268  for(itr = 0; itr < size; itr++)
269  {
270  buf[itr] = _this->fn->charAt(_this, itr);
271  }
272 
273  return size;
274 }
