237 int32_t _leFixedString_Compare(const leFixedString* _this,
238  const leString* tgt)
239 {
240  uint32_t len, itr, chr;
241 
242  LE_ASSERT_THIS();
243 
244  if(_this == NULL && tgt == NULL)
245  return 0;
246 
247  if((_this == NULL && tgt != NULL) || (_this != NULL && tgt == NULL))
248  return -1;
249 
250  len = tgt->fn->length(tgt);
251 
252  if(_this->length < len)
253  {
254  len = _this->length;
255  }
256 
257  if(len == 0)
258  return 0;
259 
260  for(itr = 0; itr < len; itr++)
261  {
262  chr = tgt->fn->charAt(tgt, itr);
263 
264  if(_this->data[itr] != chr)
265  {
266  return _this->data[itr] - chr;
267  }
268  }
269 
270  return 0;
271 }

273 leResult _leFixedString_Append(leFixedString* _this,
274  const leString* val)
275 {
276  uint32_t cpyLen, itr;
277 
278  LE_ASSERT_THIS();
279 
280  if(_this->data == NULL)
281  return LE_FAILURE;
282 
283  if(val == NULL || val->fn->length(val) == 0)
284  return LE_SUCCESS;
285 
286  cpyLen = val->fn->length(val);
287 
288  if(cpyLen >= _this->length)
289  {
290  cpyLen = _this->capacity - _this->length;
291  }
292 
293  _this->fn->preinvalidate(_this);
294 
295  for(itr = 0; itr < cpyLen; itr++)
296  {
297  _this->data[_this->length + itr] = val->fn->charAt(val, itr);
298  }
299 
300  _this->fn->invalidate(_this);
301 
302  return LE_SUCCESS;

305 leResult _leFixedString_Insert(leFixedString* _this,
306  const leString* val,
307  uint32_t idx)
308 {
309  uint32_t cpyLen;
310  int32_t itr;
311 
312  LE_ASSERT_THIS();
313 
314  if(_this->data == NULL)
315  return LE_FAILURE;
316 
317  if(idx >= _this->length)
318  {
319  return _this->fn->append(_this, val);
320  }
321 
322  if(val == NULL || val->fn->length(val) == 0)
323  return LE_SUCCESS;
324 
325  cpyLen = val->fn->length(val);
326 
327  if(cpyLen == 0)
328  return LE_SUCCESS;
329 
330  _this->fn->preinvalidate(_this);
331 
332  /* shift data right as much as possible to make room for the new values */
333  for(itr = (int32_t)_this->length - 1; itr >= (int32_t)_this->length - 1 - (int32_t)idx; itr--)
334  {
335  if(itr + cpyLen < _this->capacity)
336  {
337  _this->data[itr + cpyLen] = _this->data[itr];
338  }
339  }
340 
341  /* insert new data */
342  for(itr = 0; itr < (int32_t)cpyLen; itr++)
343  {
344  if(idx + itr >= _this->capacity)
345  break;
346 
347  _this->data[idx + itr] = val->fn->charAt(val, itr);
348  }
349 
350  _this->length += cpyLen;
351 
352  if(_this->length > _this->capacity)
353  {
354  _this->length = _this->capacity;
355  }
356 
357  _this->fn->invalidate(_this);
358 
359  return LE_SUCCESS;
360 }

362 leResult _leFixedString_Remove(leFixedString* _this,
363  uint32_t idx,
364  uint32_t count)
365 {
366  uint32_t itr;
367 
368  LE_ASSERT_THIS();
369 
370  if(_this->data == NULL || idx >= _this->length)
371  return LE_FAILURE;
372 
373  if(count == 0)
374  return LE_SUCCESS;
375 
376  _this->fn->preinvalidate(_this);
377 
378  /* simple case, just move length index */
379  if(idx + count == (uint32_t)_this->length - 1)
380  {
381  _this->length = idx;
382 
383  return LE_SUCCESS;
384  }
385 
386  if(idx + count >= (uint32_t)_this->length)
387  {
388  count = _this->length - 1 - idx;
389  }
390 
391  /* shift data left */
392  for(itr = 0; itr < count; itr++)
393  {
394  _this->data[idx + itr] = _this->data[idx + itr + 1];
395  }
396 
397  _this->length -= count;
398 
399  _this->fn->invalidate(_this);
400 
401  return LE_SUCCESS;
402 }
