115 static leResult _initReadStage(leRawDecodeState* state)
116 {
117  // initialize read stage
118 #if LE_STREAMING_ENABLED == 1
119  if(state->source->header.location == LE_STREAM_LOCATION_ID_INTERNAL)
120  {
121 #endif
122  if(_leRawImageDecoder_ReadStage_Internal(state) == LE_FAILURE)
123  {
124  return LE_FAILURE;
125  }
126 #if LE_STREAMING_ENABLED == 1
127  }
128  else
129  {
130  if(state->source->format == LE_IMAGE_FORMAT_RAW)
131  {
132  if(state->source->palette != NULL)
133  {
134  _leRawImageDecoder_ReadStage_StreamIndex(state);
135  }
136  else
137  {
138  _leRawImageDecoder_ReadStage_StreamColor(state);
139  }
140  }
141  else
142  {
143  if(state->source->palette != NULL)
144  {
145  _leRawImageDecoder_ReadStage_StreamRLEIndex(state);
146  }
147  else
148  {
149  _leRawImageDecoder_ReadStage_StreamRLE(state);
150  }
151  }
152  }
153 #endif
154 
155  return LE_SUCCESS;
156 }

217 static leResult _draw(const leImage* img,
218  const leRect* srcRect,
219  int32_t x,
220  int32_t y,
221  uint32_t a)
222 {
223  leRect imgRect, sourceClipRect, drawRect, clipRect;
224  leRect dmgRect = leRenderer_GetDrawRect();
225 
226  // only allow a new setup if there isn't a current one
227  if(state.mode != LE_RAW_MODE_NONE)
228  return LE_FAILURE;
229 
230  memset(&state, 0, sizeof(leRawDecodeState));
231 
232  imgRect.x = 0;
233  imgRect.y = 0;
234  imgRect.width = img->buffer.size.width;
235  imgRect.height = img->buffer.size.height;
236 
237  /* make sure the source rect is within the source bounds */
238  if(leRectIntersects(&imgRect, srcRect) == LE_FALSE)
239  return LE_FAILURE;
240 
241  leRectClip(&imgRect, srcRect, &sourceClipRect);
242 
243  drawRect.x = x;
244  drawRect.y = y;
245  drawRect.width = sourceClipRect.width;
246  drawRect.height = sourceClipRect.height;
247 
248  /* make sure the dest rect is within the damaged rect area */
249  clipRect = leRectClipAdj(&drawRect, &dmgRect, &sourceClipRect);
250 
251  if(sourceClipRect.width <= 0 || sourceClipRect.height <= 0)
252  return LE_FAILURE;
253 
254 #if LE_STREAMING_ENABLED == 1
255  state.manager.exec = _exec;
256  state.manager.isDone = _isDone;
257  state.manager.abort = _abort;
258  state.manager.cleanup = _cleanup;
259 #endif
260 
261  state.mode = LE_RAW_MODE_DRAW;
262 
263  state.source = img;
264  state.sourceRect = sourceClipRect;
265 
266  //state.renderX = x;
267  //state.renderY = y;
268 
269  state.destRect = clipRect;
270 
271  state.globalAlpha = a;
272 
273  state.targetMode = LE_GLOBAL_COLOR_MODE;
274 
275  if(_leRawImageDecoder_SourceIterateSetupStage(&state) == LE_FAILURE ||
276  _initReadStage(&state) == LE_FAILURE ||
277  _leRawImageDecoder_PostReadStage(&state) == LE_FAILURE ||
278  _initMaskStage(&state) == LE_FAILURE ||
279  _initLookupStage(&state) == LE_FAILURE ||
280  _initBlendStage(&state) == LE_FAILURE ||
281  _initConvertStage(&state) == LE_FAILURE ||
282  _leRawImageDecoder_FrameBufferWriteStage(&state) == LE_FAILURE)
283  {
284  return LE_FAILURE;
285  }
286 
287  return LE_SUCCESS;
288 }

290 static leResult _resize(const leImage* src,
291  const leRect* srcRect,
292  leImageFilterMode mode,
293  uint32_t sizeX,
294  uint32_t sizeY,
295  leImage* target)
296 {
297  leRect imgRect, sourceClipRect, drawRect, targetRect, clipRect;
298 
299  // only allow a new setup if there isn't a current one
300  if(state.mode != LE_RAW_MODE_NONE)
301  return LE_FAILURE;
302 
303  if(mode == LE_IMAGEFILTER_NONE)
304  return LE_FAILURE;
305 
306  // can't blend
307  if(mode == LE_IMAGEFILTER_BILINEAR &&
308  (LE_COLOR_MODE_IS_PIXEL(src->buffer.mode) == LE_FALSE ||
309  LE_COLOR_MODE_IS_PIXEL(target->buffer.mode) == LE_FALSE))
310  {
311  return LE_FAILURE;
312  }
313 
314  memset(&state, 0, sizeof(leRawDecodeState));
315 
316  /* make sure the source rect is within the source bounds */
317  imgRect.x = 0;
318  imgRect.y = 0;
319  imgRect.width = src->buffer.size.width;
320  imgRect.height = src->buffer.size.height;
321 
322  if(leRectIntersects(&imgRect, srcRect) == LE_FALSE)
323  return LE_FAILURE;
324 
325  // trim the source rectangle to fit
326  leRectClip(&imgRect, srcRect, &sourceClipRect);
327 
328  drawRect.x = 0;
329  drawRect.y = 0;
330  drawRect.width = sizeX;
331  drawRect.height = sizeY;
332 
333  targetRect.x = 0;
334  targetRect.y = 0;
335  targetRect.width = target->buffer.size.width;
336  targetRect.height = target->buffer.size.height;
337 
338  /* make sure the dest rect is within the damaged rect area */
339  clipRect = leRectClipAdj(&drawRect, &targetRect, &sourceClipRect);
340 
341  if(sourceClipRect.width <= 0 || sourceClipRect.height <= 0)
342  return LE_FAILURE;
343 
344 #if LE_STREAMING_ENABLED == 1
345  state.manager.exec = _exec;
346  state.manager.isDone = _isDone;
347  state.manager.abort = _abort;
348  state.manager.cleanup = _cleanup;
349 #endif
350 
351  state.mode = LE_RAW_MODE_RESIZE;
352 
353  state.source = src;
354  state.sourceRect = sourceClipRect;
355 
356  //state.renderX = 0;
357  //state.renderY = 0;
358 
359  state.filterMode = mode;
360  state.target = target;
361 
362  state.destRect = clipRect;
363 
364  state.targetMode = state.target->buffer.mode;
365 
366  // iterator setup
367  if(_leRawImageDecoder_TargetIterateSetupStage(&state) == LE_FAILURE)
368  return LE_FAILURE;
369 
370  // filter pre read stage
371  if(mode == LE_IMAGEFILTER_NEAREST_NEIGHBOR)
372  {
373  if(_leRawImageDecoder_ScaleNearestNeighborPreReadStage(&state) == LE_FAILURE)
374  return LE_FAILURE;
375  }
376  else
377  {
378  if(_leRawImageDecoder_ScaleBilinearPreReadStage(&state) == LE_FAILURE)
379  {
380  return LE_FAILURE;
381  }
382  }
383 
384  // read stage
385  if(_initReadStage(&state) == LE_FAILURE)
386  return LE_FAILURE;
387 
388  // post read stage
389  if(mode == LE_IMAGEFILTER_BILINEAR)
390  {
391  if(_leRawImageDecoder_ScaleBilinearPostReadStage(&state) == LE_FAILURE)
392  return LE_FAILURE;
393  }
394  else
395  {
396  if(_leRawImageDecoder_PostReadStage(&state) == LE_FAILURE)
397  return LE_FAILURE;
398  }
399 
400  // convert and write
401  if(_initConvertStage(&state) == LE_FAILURE ||
402  _leRawImageDecoder_ImageWriteStage(&state) == LE_FAILURE)
403  {
404  return LE_FAILURE;
405  }
406 
407  return LE_SUCCESS;
408 }

848  }
