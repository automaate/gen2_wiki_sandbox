97 static void nextState(leImageSequenceWidget* img)
98 {
99  switch(img->widget.drawState)
100  {
101  case NOT_STARTED:
102  {
103  paintState.alpha = 255;
104 
105 #if LE_ALPHA_BLENDING_ENABLED == 1
106  if(img->fn->getCumulativeAlphaEnabled(img) == LE_TRUE)
107  {
108  paintState.alpha = img->fn->getCumulativeAlphaAmount(img);
109  }
110 #endif
111 
112  if(img->widget.backgroundType != LE_WIDGET_BACKGROUND_NONE)
113  {
114  img->widget.drawState = DRAW_BACKGROUND;
115  img->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
116 
117  return;
118  }
119  }
120  case DRAW_BACKGROUND:
121  {
122  if(img->activeIdx >= 0 && img->activeIdx < (int32_t)img->count &&
123  img->images[img->activeIdx].image != NULL)
124  {
125  img->widget.drawState = DRAW_IMAGE;
126  img->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawImage;
127 
128  return;
129  }
130  }
131  case DRAW_IMAGE:
132  {
133  if(img->widget.borderType != LE_WIDGET_BORDER_NONE)
134  {
135  img->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
136  img->widget.drawState = DRAW_BORDER;
137 
138  return;
139  }
140  }
141  case DRAW_BORDER:
142  {
143  img->widget.drawState = DONE;
144  img->widget.drawFunc = NULL;
145  }
146  }
147 }
