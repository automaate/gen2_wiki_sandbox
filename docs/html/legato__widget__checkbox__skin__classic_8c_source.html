104  }

197 static void nextState(leCheckBoxWidget* cbox)
198 {
199  switch(cbox->widget.drawState)
200  {
201  case NOT_STARTED:
202  {
203  paintState.alpha = 255;
204 
205 #if LE_ALPHA_BLENDING_ENABLED == 1
206  if(cbox->fn->getCumulativeAlphaEnabled(cbox) == LE_TRUE)
207  {
208  paintState.alpha = cbox->fn->getCumulativeAlphaAmount(cbox);
209  }
210 #endif
211 
212  if(cbox->widget.backgroundType != LE_WIDGET_BACKGROUND_NONE)
213  {
214  cbox->widget.drawState = DRAW_BACKGROUND;
215  cbox->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
216 
217  return;
218  }
219  }
220  case DRAW_BACKGROUND:
221  {
222  cbox->widget.drawState = DRAW_IMAGE;
223  cbox->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawImage;
224 
225  return;
226  }
227  case DRAW_IMAGE:
228  {
229  if(cbox->string != NULL &&
230  cbox->string->fn->length(cbox->string) != 0)
231  {
232  cbox->widget.drawState = DRAW_STRING;
233  cbox->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawString;
234 
235  return;
236  }
237  }
238  case DRAW_STRING:
239  {
240  if(cbox->widget.borderType != LE_WIDGET_BORDER_NONE)
241  {
242  cbox->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
243  cbox->widget.drawState = DRAW_BORDER;
244 
245  return;
246  }
247  }
248  case DRAW_BORDER:
249  {
250  cbox->widget.drawState = DONE;
251  cbox->widget.drawFunc = NULL;
252  }
253  }
254 }

343 static void drawImage(leCheckBoxWidget* cbox)
344 {

447 void _leCheckBoxWidget_Paint(leCheckBoxWidget* cbox)
448 {
449  if(cbox->widget.scheme == NULL)
450  {
451  cbox->widget.drawState = DONE;
452 
453  return;
454  }
455 
456  if(cbox->widget.drawState == NOT_STARTED)
457  {
458  nextState(cbox);
459  }
460 
461 #if LE_STREAMING_ENABLED == 1
462  if(cbox->widget.drawState == WAIT_IMAGE ||
463  cbox->widget.drawState == WAIT_STRING)
464  {
465  return;
466  }
467 #endif
468 
469  while(cbox->widget.drawState != DONE)
470  {
471  cbox->widget.drawFunc((leWidget*)cbox);
472 
473 #if LE_PREEMPTION_LEVEL == 2
474  break;
475 #endif
476 
477 #if LE_STREAMING_ENABLED == 1
478  if(cbox->widget.drawState == WAIT_IMAGE ||
479  cbox->widget.drawState == WAIT_STRING)
480  break;
481 #endif
482  }
