228 leResult leInitialize(const gfxDisplayDriver* dispDriver)
229 {
230  uint32_t idx;
231  leWidget* root;
232 
233  if(_initialized == LE_TRUE)
234  return LE_FAILURE;
235 
236  memset(&_state, 0, sizeof(leState));
237 
238  if(leMemory_Init() == LE_FAILURE ||
239  leEvent_Init() == LE_FAILURE ||
240  leInput_Init() == LE_FAILURE ||
241  leRenderer_Initialize(dispDriver) == LE_FAILURE)
242  {
243  return LE_FAILURE;
244  }
245 
246 #if LE_DYNAMIC_VTABLES == 1
247  /* initialize virtual function tables */
248  idx = 0;
249 
250  while(vtableFnTable[idx] != NULL)
251  {
252  vtableFnTable[idx]();
253 
254  idx += 1;
255  }
256 #endif
257 
258  leImage_InitDecoders();
259  leScheme_Initialize(&_state.defaultScheme, LE_GLOBAL_COLOR_MODE);
260 
261  for(idx = 0; idx < LE_LAYER_COUNT; idx++)
262  {
263  root = &_state.rootWidget[idx];
264 
265  leWidget_Constructor(root);
266 
267  root->fn->setPosition(root, 0, 0);
268 
269  root->fn->setSize(root,
270  dispDriver->getDisplayWidth(),
271  dispDriver->getDisplayHeight());
272  }
273 
274  _initialized = LE_TRUE;
275 
276  return LE_SUCCESS;
277 }

300  _initialized = LE_FALSE;
301 }

304 {

319 {

332 leResult leUpdate(uint32_t dt)
333 {
334  leEvent_ProcessEvents();
335 
336 #if LE_STREAMING_ENABLED == 1
337  // there is an active stream in progress, service that to completion
338  // before painting anything else
339  if(_state.activeStream != NULL)
340  {
341  if(_state.activeStream->isDone(_state.activeStream) == LE_TRUE)
342  {
343  _state.activeStream = NULL;
344  }
345  else
346  {
347  _state.activeStream->exec(_state.activeStream);
348 
349  return LE_SUCCESS;
350  }
351  }
352 #endif
353 
354  updateWidgets(dt);
355 
356  leRenderer_Paint();
357 
358  return LE_SUCCESS;
359 }
