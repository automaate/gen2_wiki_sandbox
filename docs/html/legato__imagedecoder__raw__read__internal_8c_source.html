159 leResult _leRawImageDecoder_ReadStage_Internal(leRawDecodeState* state)
160 {
161  memset(&internalReadStage, 0, sizeof(internalReadStage));
162 
163  internalReadStage.dataSize = leColorInfoTable[state->source->buffer.mode].size;
164  internalReadStage.imgBPP = leColorInfoTable[state->source->buffer.mode].bppOrdinal;
165 
166  internalReadStage.base.state = state;
167 
168  if(LE_COLOR_MODE_IS_INDEX(state->source->buffer.mode) == LE_TRUE)
169  {
170  if(state->source->format == LE_IMAGE_FORMAT_RAW)
171  {
172  internalReadStage.decode = indexDecode;
173  }
174  else
175  {
176  if(state->randomRLE == LE_FALSE)
177  {
178  internalReadStage.decode = indexRLESequentialDecode;
179  }
180  else
181  {
182  internalReadStage.decode = indexRLERandomDecode;
183  }
184  }
185  }
186  else
187  {
188  if(state->source->format == LE_IMAGE_FORMAT_RAW)
189  {
190  internalReadStage.decode = colorDecode;
191  }
192  else
193  {
194  if(state->randomRLE == LE_FALSE)
195  {
196  internalReadStage.decode = rleSequentialDecode;
197  }
198  else
199  {
200  internalReadStage.decode = rleRandomDecode;
201  }
202  }
203  }
204 
205  internalReadStage.base.exec = (leResult(*)(struct leRawDecodeStage*)) stage_decode;
206 
207  _leRawImageDecoder_InjectStage(state, (void*)&internalReadStage);
208 
209  return LE_SUCCESS;
