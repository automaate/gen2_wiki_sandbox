135 /* Enumeration:
136  leImageSequenceWidget
137 
138  Summary:
139  Image sequence widget struct definition
140 
141  Description:
142  An image sequence widget is similar to an image widget with the additional
143  capability of showing a sequence of images and automating the transition
144  between them.
145 
146  This widget is dependent on the time value provided to leUpdate. If
147  leUpdate is not provided with time information this widget will not be
148  able to automatically cycle.
149 
150  Remarks:
151  None.
152 */
153 typedef struct leImageSequenceWidget
154 {
155  leWidget widget; // widget base class
156 
157  const leImageSequenceWidgetVTable* fn;
158 
159  uint32_t count; // number of image entries for this widget
160  leImageSequenceEntry* images; // image entry array
161 
162  int32_t activeIdx; // currently displayed entry
163 
164  leBool playing; // indicates that the widget is automatically cycling
165  uint32_t time; // current cycle time
166 
167  leBool repeat; // indicates that the sequence should repeat when it
168  // reaches the end of the sequence
169 
170  leImageSequenceImageChangedEvent_FnPtr cb; // callback when the image changes
171 } leImageSequenceWidget;
