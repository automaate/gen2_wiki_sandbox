447 leResult leRectArray_CropToSize(leRectArray* arr, uint32_t size)
448 {
449  uint32_t rectItr;
450  leRect split;
451 
452  /* minimal sane magic number size limit */
453  if(size < 4)
454  return LE_FAILURE;
455 
456  for(rectItr = 0; rectItr < arr->size; rectItr++)
457  {
458  while((uint32_t)arr->rects[rectItr].width *
459  (uint32_t)arr->rects[rectItr].height > size)
460  {
461  split = arr->rects[rectItr];
462 
463 #if LE_RENDER_LEFTRIGHT == 0
464  // split favoring height if possible
465  if((uint32_t)arr->rects[rectItr].height <= 2)
466  {
467  arr->rects[rectItr].width >>= 1;
468 
469  split.width -= arr->rects[rectItr].width;
470  split.x += arr->rects[rectItr].width;
471  }
472  else

474  arr->rects[rectItr].height >>= 1;
475 
476  split.height -= arr->rects[rectItr].height;
477  split.y += arr->rects[rectItr].height;
