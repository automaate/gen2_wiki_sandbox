43 leResult leFixedHeap_Init(leFixedHeap* heap,
44  uint32_t size,
45  uint32_t count,
46  uint8_t* data)
47 {
48  leFixedHeapBlock* block;
49  uint32_t i;
50 
51 #if LE_FIXEDHEAP_DEBUG == 1
52  uint32_t* checksum;
53 #endif
54 
55  memset(heap, 0, sizeof(leFixedHeap));
56 
57  // validate args
58  if(heap == NULL || count == 0)
59  return LE_FAILURE;
60 
61  if(size <= sizeof(void*))
62  {
63  size = sizeof(void*);
64  }
65 
66  // set up sizes
67  heap->numElements = count;
68  heap->logicalBlockSize = size;
69 
70  heap->physicalBlockSize = LE_FIXEDHEAP_BLOCK_SIZE(heap->logicalBlockSize);
71 
72  // allocate memory
73  heap->data = data;
74 
75  LE_ASSERT(heap->data != NULL);
76 
77  //printf("fixed heap: 0x%08X, %u, %u\n", (uint32_t)heap->data, heap->numElements, heap->logicalBlockSize);
78 
79  if(heap->data == NULL)
80  return LE_FAILURE;
81 
82 #if LE_FIXEDHEAP_DEBUG == 1
83  memset(heap->data, 0, heap->physicalBlockSize * heap->numElements);
84 #endif
85 
86  // init free list
87  heap->freeList = heap->data;
88 
89  block = (leFixedHeapBlock*)heap->data;
90 
91  for (i = 0; i < heap->numElements; i++)
92  {
93 #if LE_FIXEDHEAP_DEBUG == 1
94  block->debug.checksum = (uint32_t)block ^ LE_FIXEDHEAP_CHECKSUM;
95  checksum = (uint32_t*)((uint8_t*)block + heap->physicalBlockSize - LE_FIXEDHEAP_FOOTER_SIZE);
96  *checksum = block->debug.checksum;
97 
98  block->debug.free = LE_TRUE;
99 #endif
100 
101  if(i < heap->numElements - 1)
102  {
103  block->data = (void*)((uint8_t*)block + heap->physicalBlockSize);
104  block = (leFixedHeapBlock*)(((uint8_t*)block) + heap->physicalBlockSize);
105  }
106  else
107  {
108  block->data = NULL;
109  }
110  }
111 
112  heap->capacity = count;
113  heap->maxUsage = 0;
114  heap->initialized = LE_TRUE;
115 
116  return LE_SUCCESS;
117 }

130 void* leFixedHeap_Alloc(leFixedHeap* heap,
131  uint32_t lineNum,
132  const char* funcName,

134 #else
135 void* leFixedHeap_Alloc(leFixedHeap* heap)
136 #endif
137 {
138  leFixedHeapBlock* freeBlk;
139  leFixedHeapBlock* allocBlk;
140 
141  LE_ASSERT(heap != NULL && heap->initialized == LE_TRUE);
142 
143  LE_ASSERT_VAR(heap->capacity != 0 &&
144  heap->capacity <= heap->numElements &&
145  heap->freeList != NULL, "Fixed heap %u is full", heap->logicalBlockSize);
146 
147  if(heap->freeList == NULL)
148  return NULL;
149 
150  freeBlk = (leFixedHeapBlock*)heap->freeList;
151 
152  heap->freeList = freeBlk->data;
153 
154  allocBlk = (leFixedHeapBlock*)freeBlk;
155 
156 #if LE_FIXEDHEAP_DEBUG == 1
157 #if LE_USE_ALLOCATION_TRACKING == 1
158  allocBlk->debug.lineNumber = lineNum;
159  allocBlk->debug.funcName = funcName;
160  allocBlk->debug.fileName = fileName;
161 #endif
162 
163  allocBlk->debug.free = LE_FALSE;
164 
165  allocBlk->data = NULL;
166 
167  memset(&allocBlk->data, MS_ALLOC, heap->logicalBlockSize);
168 #endif
169 
170  heap->capacity--;
171 
172  if(heap->capacity == 0)
173  {
174  heap->freeList = NULL;
175  }
176 
177 #if LE_FIXEDHEAP_DEBUG == 1
178  LE_ASSERT(leFixedHeap_Validate(heap) == LE_SUCCESS);
179 #endif
180 
181  if(heap->numElements - heap->capacity > heap->maxUsage)
182  {
183  heap->maxUsage = heap->numElements - heap->capacity;
184  }
185 
186  return &allocBlk->data;
187 }

189 void leFixedHeap_Free(leFixedHeap* heap,
190  void* ptr)
191 {
192  leFixedHeapBlock *oldFreeBlk, *freeBlk;
193 
194  LE_ASSERT(heap != NULL && heap->initialized == LE_TRUE);
195 
196  LE_ASSERT_VAR(heap->capacity != heap->numElements, "Trying to free from an empty heap (%u)", heap->logicalBlockSize);
197 
198 #if LE_FIXEDHEAP_DEBUG == 1
199  LE_ASSERT(leFixedHeap_Validate(heap) == LE_SUCCESS);
200 #endif
201 
202  oldFreeBlk = (leFixedHeapBlock*)heap->freeList;
203  freeBlk = (leFixedHeapBlock*)((uint8_t*)ptr - LE_FIXEDHEAP_HEADER_SIZE);
204 
205 #if LE_FIXEDHEAP_DEBUG == 1
206  LE_ASSERT_MSG(freeBlk->debug.free != LE_TRUE, "Trying to free an address that hasn't been allocated");
207 
208  memset(&freeBlk->data, MS_FREE, heap->logicalBlockSize);
209 
210  freeBlk->debug.free = LE_TRUE;
211 #endif
212 
213  freeBlk->data = (void*)oldFreeBlk;
214  heap->freeList = (void*)freeBlk;
215 
216  //_ASSERT(KT_FixedHeap_Contains(heap, heap->freeList) == KT_TRUE);
217 
218  heap->capacity++;
219 
220 #if LE_FIXEDHEAP_DEBUG == 1
221  LE_ASSERT(leFixedHeap_Validate(heap) == LE_SUCCESS);
222 #endif
223 }

225 leBool leFixedHeap_Contains(leFixedHeap* heap, void* ptr)
226 {
227  uint8_t *data, *dataPtr;
228 
229  LE_ASSERT(heap != NULL && heap->initialized == LE_TRUE);
230 
231  if(heap->data == NULL)
232  return LE_FALSE;
233 
234  data = (uint8_t*)heap->data;
235  dataPtr = (uint8_t*)ptr;
236 
237  if(dataPtr < data || dataPtr >= data + (heap->physicalBlockSize * heap->numElements))
238  return LE_FALSE;
239 
240  dataPtr -= (uint32_t)data;
241 
242  return ((uint32_t)dataPtr % heap->physicalBlockSize) == LE_FIXEDHEAP_HEADER_SIZE;
243 }

245 leResult leFixedHeap_Validate(struct leFixedHeap* heap)
246 {
247 #if LE_FIXEDHEAP_DEBUG == 1
248  leFixedHeapBlock* blk;
249  uint32_t* ftr;
250  uint32_t i;
251 
252  blk = heap->data;
253 
254  for(i = 0; i < heap->numElements; i++)
255  {
256  // validate checksums
257  if(blk->debug.checksum != ((uint32_t)blk ^ LE_FIXEDHEAP_CHECKSUM))
258  return LE_FAILURE;
259 
260  ftr = (uint32_t*)((uint8_t*)blk + heap->physicalBlockSize - sizeof(uint32_t));
261 
262  if(*ftr != blk->debug.checksum)
263  return LE_FAILURE;
264  }
265 
266  i = 0;
267 
268  while(1)
269  {
270  if(i == heap->numElements - heap->capacity)
271  break;
272 
273  blk = heap->freeList;
274 
275  blk = (leFixedHeapBlock*)blk->data;
276 
277  i++;
278  }
279 #endif
280 
281  return LE_SUCCESS;
282 }
