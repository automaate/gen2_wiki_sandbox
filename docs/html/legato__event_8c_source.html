91 {

95 
96  if(_state.events.size == 0)

100 
101  if(_state.events.size > 1)
102  {
103  node = node;
104  }
105 
106  while(node != NULL)
107  {
108  //evt = _state.events.values[i];
109  evt = (leEvent*)node->val;
110 
111  if(_state.filter != NULL &&
112  _state.filter(evt) == LE_FALSE)
113  continue;
114 
115  node = node->next;
116 
117  res = processEvent(evt);
118 
119  if(res == LE_EVENT_HANDLED)
120  {
121  leList_Remove(&_state.events, evt);
122 
123  LE_FREE(evt);
124  }
125  else if(res == LE_EVENT_RESET_QUEUE)
126  {
127  leList_Destroy(&_state.events);
128 
129  node = NULL;
130  }
131  }
132 
133  return LE_SUCCESS;
134 }
