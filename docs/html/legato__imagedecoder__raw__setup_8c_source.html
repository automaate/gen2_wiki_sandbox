37 static leResult stage_sourceIterateSetup(leRawDecodeStage* stage)
38 {
39  leRawDecodeState* state = stage->state;
40 
41  if(state->rowIterator >= (uint32_t)state->sourceRect.height)
42  {
43  state->currentStage = 0;
44  state->done = LE_TRUE;
45 
46  return LE_SUCCESS;
47  }
48 
49  state->referenceX = state->sourceRect.x + state->colIterator;
50  state->referenceY = state->sourceRect.y + state->rowIterator;
51 
52  // calculate target coordinates
53  state->targetX = state->destRect.x + state->colIterator;
54  state->targetY = state->destRect.y + state->rowIterator;
55 
56  state->readCount = 1;
57 
58  state->readOperation[0].x = state->sourceRect.x + state->colIterator;
59  state->readOperation[0].y = state->sourceRect.y + state->rowIterator;
60 
61  // calculate buffer index
62  state->readOperation[0].bufferIndex = state->readOperation[0].x +
63  (state->readOperation[0].y *
64  stage->state->source->buffer.size.width);
65 
66  if(state->colIterator < (uint32_t)state->sourceRect.width - 1)
67  {
68  state->colIterator += 1;
69  }
70  else
71  {
72  state->colIterator = 0;
73  state->rowIterator += 1;
74  }
75 
76  return LE_SUCCESS;
77 }
