31 uint32_t leGetRLEDataAtIndex(uint8_t* data,
32  uint32_t max,
33  uint32_t idx,
34  uint32_t* startBlock,
35  uint32_t* startOffset)
36 {
37  uint32_t rleRunLength;
38  uint32_t rleDataSize;
39  uint32_t length = 0;
40  uint32_t val = 0;
41  uint32_t offset = 0;
42  uint32_t i = 0;
43 
44  rleRunLength = (uint32_t)data[0];
45  rleDataSize = (uint32_t)data[1];
46 
47  data += RLE_HEADER_SIZE + ((*startBlock) * (rleRunLength + rleDataSize));
48  offset = *startOffset;
49 
50  while(offset < max)
51  {
52  // get the length
53  for(i = 0; i < rleRunLength; i++)
54  {
55  ((uint8_t*)&length)[i] = data[i];
56  }
57 
58  // index is in this range, return value
59  if(idx < offset + length)
60  {
61  // get the value
62  for(i = 0; i < rleDataSize; i++)
63  {
64  ((uint8_t*)&val)[i] = data[i + rleRunLength];
65  }
66 
67  return val;
68  }
69 
70  data += rleRunLength + rleDataSize;
71 
72  offset += length;
73 
74  *startBlock += 1;
75  *startOffset = offset;
76  }
77 
78  return 0;
79 }
